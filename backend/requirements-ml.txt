# Heavy ML dependencies (Phase 2+). Kept out of requirements.txt so the web
# skeleton installs in seconds instead of pulling ~3 GB.
#
# TORCH MUST BE INSTALLED FIRST, from the CUDA-specific index — the default PyPI
# wheel is a different build and pip will not "upgrade" you onto the right one
# once a plain torch is resident. PICK THE INDEX MATCHING YOUR OWN GPU AND
# DRIVER from https://pytorch.org/get-started/locally/ — the cu126 URL below is
# an example, not a recommendation for your machine:
#
#     pip install torch torchvision --index-url https://download.pytorch.org/whl/cu126
#     pip install -r requirements-ml.txt
#
# Verify afterwards with:
#     python -c "import torch; print(torch.cuda.is_available(), torch.cuda.get_device_name(0))"

# --- Auto-annotation ---
# transformers ships Grounding DINO (tiny + base), OWLv2 and Florence-2 — four
# of the five annotators from one dependency and one weight cache. Avoids
# vendoring the original research repos, which are pinned to old torch versions
# and are painful to install on Windows.
transformers>=4.46.0
timm>=1.0.0
accelerate>=1.0.0

# YOLO-World's CLIP text encoder. Without this, ultralytics tries to
# auto-install its CLIP fork from GitHub at first use — which needs git on the
# machine. The PyPI republication provides the same importable `clip` module,
# so machines without git still work.
openai-clip>=1.0.1

# Box/mask utilities and NMS. Also used in Phase 5 to draw predictions.
supervision>=0.25.0

# --- Phase 4: training ---
# YOLO11 (Ultralytics) — the first training backend (app/ml/trainers/yolo.py).
# Pulls polars + nvidia-ml-py + thop; torch is already satisfied, so the CUDA
# build is left untouched. Nano fine-tunes on modest GPUs.
ultralytics>=8.3.0
# RF-DETR (app/ml/trainers/rf_detr.py) — Roboflow's detection transformer.
# The [train] extras pull the PyTorch Lightning training stack; inference
# alone would not need them, but training is the whole point here.
rfdetr[train]>=1.7.0

# ONNX export (GET /api/models/{id}/onnx). ultralytics would auto-install this
# at first export; declaring it means the feature works offline too.
onnx>=1.16.0

# --- Phase 5: evaluation ---
# COCO mAP on a held-out test split (services/evaluation_job.py). The reference
# implementation of detection mAP — a prebuilt Windows wheel now exists, so it
# no longer needs a C toolchain to install.
pycocotools>=2.0.8


# =============================================================================
# VRAM BUDGET — what each model actually costs
# =============================================================================
# Available VRAM is the binding constraint on model choice, and it is why
# app/ml/registry.py enforces one resident model at a time and
# services/gpu_admission.py meters concurrent jobs. These figures are
# properties of the MODELS — compare them against whatever card you run on.
# The app reports yours in the Compute panel, and each picker shows the
# per-variant estimate beside the size.
#
#   Grounding DINO tiny     ~700 MB weights   ~2.5 GB inference
#   Grounding DINO base     ~1.8 GB weights   ~5.5 GB inference
#   YOLO-World S             ~25 MB weights   ~2 GB inference
#   OWLv2 base              ~1.2 GB weights   ~3 GB inference
#   Florence-2 base         ~500 MB weights   ~2.5 GB inference
#
# Training wants far more than inference — the per-variant estimates live on
# the trainer classes (app/ml/trainers/) and drive both the adaptive batch
# default and the GPU-admission reservations.
